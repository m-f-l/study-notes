    # null和undefined

     let u: undefined = undefined 
 let n: null = null

    注意：：与void的区别是，undefined 和null 是所有类型的子类型。也就是说undefined 类型的变量，可以赋值给其他类型的变量，例如number

    let u :undefined; let num: number = u 
// 不会报错（前提是非严格模式下）

    let u :void; let num: number = u 
//严格模式下报错： Type 'void' is not assignable to type 'number'

    严格模式下的严格模式配置：

    {
  "compilerOptions": {
    "strictNullChecks":true // 在严格模式下，null和undefined 值不包括在任何类型里，只允许赋值给void和本身对应的类型
  }
}

    严格模式下 undefined ,null 不能给其他类型赋值，只能给他们自己的类型赋值，不过undefined 可以void赋值（null不可以）

    let a: null = null; 
let b:undefined = undefined; 
let c:void = undefined;

    # 数组类型

    定义数组类型：

    • 在元素类型后面加上[]，表示由此类型组成的一个数组
    let arr: number[] = [1,2,3]
    • 使用泛型
    ler arr: Array<number> = [1,2,3]
    • 使用接口interface(一般不用)
    interface NumberArray {
  [index:number]:number;
}
let number: NumberArray =[1,2,4]

    # 元组

    • 概念：元组表示已知数量和类型的数组，各元素的类型不必相同，对应位置需要相同
    • 例子：👇
    let x: [string, number]
x = ['zs', 18]
y = [18, 'zs'] //error

    # 联合类型

    • 概念：表示取值可以是多种类型中的一种。联合类型使用 | 分隔每个类型
    let a: string | number  => 允许a的类型是字符串或者数字，其他类型不允许
a = 1
a = '1'
a = true //Type 'boolean' is not assignale to tpe 'string | number'
    • 访问联合类型中的属性或方法 当ts不确定联合类型的变量到底是哪一个类型的时候，我们只能访问联合类型的所有类型里面共有的属性和方法
    function getLength(value:string| number) {
  console.log(value.length)
}
// Property 'length' does not exist on type 'number'

    number 没有length 报错

    • 解决办法
    function getLength(value:string| number[]) {
  console.log(value.length)
}
    1. 使用断言(后面会讲到)
    function getLength(value: string | number) {
    console.log((value as string).length)
}

    # 接口 interface 常用于定义对象或者函数类型

    • 基本用法
    interface Person {
  name:string;
  age: number
}

let person:Person = {
  name:'zs',
  age:18
}

    说明： 定义了一个接口Person和一个变量person 。变量person的类型是Person 。注：定义的变量的形状必须和接口的形状保持一致（即定义的变量比接口数量少或者多都是不允许的）

    • 可选属性 说明： 可选属性使用 ?: 表示 age?:number 表示age是一个可选属性，在定义了person变量的时候表示age可选，有也可以，没有也行
    interface Person {
  name:string;
  age?: number
}
let person:Person = {
  name:'zs'
}
    • 任意属性 允许一个接口有任何属性
    interface Person {
  name:string;
  age: number;
  [propName:string]:any;
}

let person: Person  = {
  name: 'zs',
  age: 18,
  sayHi:()=> {
    alert('hello world')
  }
}

    注：一旦定义了任意属性，那么确定属性和可选属性的类型必须是它的类型的子集

    interface Person{
   name:string;
   age?: number;
  [propName:string]:string;
}

let tom:Person= {
  name: 'zs',
  age: 18,
  gender: 'male'
}
//Type '{ name: string; age: number; gender: string; }' is not assignable to type 'Person'Property 'age' is incompatible with index signature.Type 'number' is not assignable to type 'string'.

    可选属是number类型，不是任意属性类型的子集

    • 可读属性 可以用readonly 定义只读属性
    interface Person {
  readonly id: number;
  name: string
}
let person: Person = {
  id:1,
  name:'zs'
}
person.id = 2
// Cannot assign to 'id' because it is a read-only property.

    id 为只读属性，后面不允许修改

    • 定义函数类型 接收了num1 number类型和num2 number类型，反悔了number类型
    interface AddType {
  (num1:number,num2:number):number
} 
let add: AddType = (num1, num2) => {
  return num1 + num2
}
    • 定义数组（一般不用）
    interface Arr {
  [index: number]: {name:string; age:number}
}
let arr: Arr = [{name:'zs',age:18}]

    # 类型别名

    类型别名，可以用类型别名为任何类型命名

    type Name = string  // 原始类型
type UnionType = string|number // 联合类型
type FnType = (num1: number, num2:number) => number  //函数
type TupleType = [number,string] // 元组

    # 对比类型别名（type）和接口（interface）

    相同点：

    1. 都可以定义一个对象或者函数 定义对象： 👇
    interface Person {
  name: string
  age: number
}

type Person = {
  name: string
  age: number
}

const zs: Person = {
  name: 'zs',
  age: 18
}

    定义函数：👇

    type addType=(num1:number,num2:number) => number

interface addType {
  (num1:number, num2:number):number
}
const add: addType = (num1, num2)=> {
  return num1 + num2
}
    1. 都允许扩展

    不同点:

    1. type类型别名可以为任何类型命名。（可以用来表示基本类型，对象类型，联合类型，元组...）,而interface 一般用来定义对象和函数

    2. interface 可以合并重复声明，type不可以

    interface Person {
  name:string
}
interface Person {
  age:number
}
let user:Person = {
  name:'zs',
  age:18
}
// 此时Person 类型相当于 {name:string; age:number}

    再看看type 👇

    type Person = {name:string}
type Person = {age:number} 
// Duplicate identifier 'Person'.

    1. type可以使用typeof 获取实例的类型进行赋值

    2. 扩展方式不同。interface扩展接口：可以通过其他接口来扩展。可以在interface后面添加关键字extends去扩展接口。类型别名通过交叉点扩展类型。type需要使用&符号去扩展接口。 interface: 👇

    interface MyInterface {
  name: string;
  say(): void;
}
interface MyInterface2 extends MyInterface {
  age: number;
}

let value: MyInterface2 = {
  name: 'zs',
  age: 18,
  say() {
    console.log('hello')
  }
}

    type: 👇

    type MyType = {
  name: string;
  say(): void;
}

type MyType2 = MyType & {
  age:number
}

let value:MyType2 = {
  name:'zs',
  age:18,
  say() {
    console.log('hello')
  }
}
